Solve linear FEA system [K]{u} = {F}

# Python import numpy as np K = np.array([[10, -2], [-2, 5]]) F = np.array([100, 0]) u = np.linalg.solve(K, F) print("Displacements:", u)
% MATLAB K = [10 -2; -2 5]; F = [100; 0]; u = K \ F; disp('Displacements:'); disp(u)

2-DOF eigenvalue — natural frequencies

import numpy as np from scipy.linalg import eig m1, m2 = 2.0, 1.5 k1, k2, k3 = 5000, 3000, 2000 M = np.diag([m1, m2]) K = np.array([[k1+k2, -k2], [-k2, k2+k3]]) vals, vecs = eig(K, M) freqs_hz = np.sqrt(vals.real) / (2*np.pi) for i, f in enumerate(sorted(freqs_hz)): print(f"Mode {i+1}: {f:.3f} Hz")

SDOF forced vibration — frequency sweep

import numpy as np import matplotlib.pyplot as plt m, k, zeta = 10, 4000, 0.05 F0 = 100 wn = np.sqrt(k/m) r = np.linspace(0.01, 3, 1000) MF = 1 / np.sqrt((1-r**2)**2 + (2*zeta*r)**2) plt.figure(figsize=(8, 4)) plt.plot(r, MF, color='#4da3ff', lw=2) plt.axvline(x=1, color='#ff6b8a', ls='--', lw=1, label='Resonance r=1') plt.axhline(y=1/(2*zeta), color='#ffc96b', ls='--', lw=1, label=f'Peak MF = {1/(2*zeta):.1f}') plt.xlabel('Frequency ratio r = ω/ωₙ') plt.ylabel('Magnification factor MF') plt.title(f'SDOF Frequency Response — ζ={zeta}') plt.legend(); plt.grid(True, alpha=0.3); plt.tight_layout(); plt.show()

Reynolds number and flow regime

# Air at 15°C over a flat plate rho = 1.225; mu = 1.789e-5; V = 30; L = 0.5 Re = rho * V * L / mu print(f"Re = {Re:.3e}") regime = "Laminar" if Re < 2300 else ("Transitional" if Re < 4000 else "Turbulent") print(f"Regime: {regime}") # Flat plate: laminar BL thickness and skin friction x = 0.3 # position along plate [m] Rex = rho * V * x / mu delta = 5 * x / Rex**0.5 Cf = 0.664 / Rex**0.5 print(f"At x={x} m: δ = {delta*1000:.3f} mm, Cf = {Cf:.5f}")

1D thermal resistance network

layers = [ {'name': 'Steel', 'k': 50, 'L': 0.01}, {'name': 'Insulation', 'k': 0.04, 'L': 0.05}, {'name': 'Concrete', 'k': 1.7, 'L': 0.20}, ] A = 1.0; T_hot = 200; T_cold = 20; h_in = 50; h_out = 10 R_conv_in = 1/(h_in*A) R_conv_out = 1/(h_out*A) R_cond = sum(l['L']/(l['k']*A) for l in layers) R_total = R_conv_in + R_cond + R_conv_out Q_dot = (T_hot - T_cold) / R_total print(f"R_total = {R_total:.4f} K/W | Q̇ = {Q_dot:.2f} W")
